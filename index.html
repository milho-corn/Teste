let player = {}; let enemy = { hp: 100, speed: 6 }; let enemySkip = false; let playerTurn = true; /* PERSONAGENS */ function selectChar(char) { if (char === "kael") { player = { name: "Kael", hp: 100, atk: 9, mag: 4, speed: 8 }; } if (char === "lyra") { player = { name: "Lyra", hp: 100, atk: 5, mag: 7, speed: 9 }; } document.getElementById("menu").classList.add("hidden"); document.getElementById("game").classList.remove("hidden"); startGame(); } function startGame() { document.getElementById("charName").innerText = player.name; // quem começa? playerTurn = player.speed >= enemy.speed; updateUI(); renderActions(); } /* ATUALIZA UI */ function updateUI() { document.getElementById("playerHpBar").style.width = player.hp + "%"; document.getElementById("enemyHpBar").style.width = enemy.hp + "%"; document.getElementById("status").innerText = (playerTurn ? "Sua vez ⚔️" : "Vez do inimigo 👾") + " | Seu HP: " + player.hp + " | Inimigo HP: " + enemy.hp; } /* CONTROLE DE TURNOS */ function nextTurn() { playerTurn = !playerTurn; if (!playerTurn) { setTimeout(enemyTurn, 600); } } /* AÇÕES KAEL */ function doubleSlash() { if (!playerTurn) return; enemy.hp -= player.atk * 2; enemy.hp -= player.atk * 2; updateUI(); checkEnd(); nextTurn(); } function smoke() { if (!playerTurn) return; enemySkip = true; document.getElementById("status").innerText = "Kael ficou invisível!"; nextTurn(); } function ultimateKael() { if (!playerTurn) return; enemy.hp -= player.atk * 5; updateUI(); checkEnd(); nextTurn(); } /* AÇÕES LYRA */ function windAttack() { if (!playerTurn) return; enemy.hp -= player.mag * 2; updateUI(); checkEnd(); nextTurn(); } function shield() { if (!playerTurn) return; player.hp += 15; updateUI(); nextTurn(); } function ultimateLyra() { if (!playerTurn) return; enemy.hp -= player.mag * 6; updateUI(); checkEnd(); nextTurn(); } /* INIMIGO */ function enemyTurn() { if (enemySkip) { enemySkip = false; playerTurn = true; return; } let dmg = Math.floor(Math.random() * 10) + 5; player.hp -= dmg; checkEnd(); playerTurn = true; updateUI(); } /* FIM DE JOGO */ function checkEnd() { if (player.hp <= 0) { alert("Você perdeu!"); location.reload(); } if (enemy.hp <= 0) { alert("Você venceu!"); location.reload(); } }
